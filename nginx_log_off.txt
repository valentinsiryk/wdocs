=============
NGINX log off
=============

# Выключить логирование доступа к файлам изображений в access_log

location ~* \.(gif|jpg|png) {
	access_log off;
}


# Сохранять логи доступа к файлам изображений в другое место

location ~* \.(gif|jpg|png)$ {
	access_log /path/to/access.log;
}


# Выключить логирование доступа к папке и подпапкам

location ^~ /images/ {
	access_log off;
}
